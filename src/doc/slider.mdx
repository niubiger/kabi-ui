import { Props, Playground } from 'docz';

# Slider

## Slider Props

<Props of={Slider} />

## Slider 示例
<Playground>
    {() => {
        const step = 10;
        const max = 100;
        const min = 0;
        const [val, setVal] = React.useState(0);
        const handleChange = React.useCallback(val => {
            console.log('change when mousemove', val);
            setVal(val);
        }, []);
        const handleChangeSubmitted = React.useCallback(val => {
            console.log('change commited after mouseup', val);
        }, []);
        return (
            <div style={{ display: 'flex', alignItems: 'center', padding: 20 }}>
                <div style={{ width: 200, display: 'flex' }}>
                    <Slider
                        value={val}
                        max={max}
                        min={min}
                        step={step}
                        onChange={handleChange}
                        onChangeCommited={handleChangeSubmitted}
                    />
                </div>
                <div style={{ marginLeft: 20, width: 100 }}>
                    <Input label="value" outerStyle={{ width: '100%' }} value={val} onChange={e => setVal(e.target.value)} type="number" step={step} max={max} min={min} />
                </div>
            </div>
        );
    }}
</Playground>
